# —————————————— scripts/quality_playability.py ——————————————
import random, json, math
from dataclasses import dataclass

@dataclass
class PlayabilityScore:
    base_winrate: float        # 随机通关率
    strategy_winrate: float    # 策略通关率
    floor_gradient: float      # 难度梯度平滑度
    choice_diversity: float    # 选择多样性
    game_length_min: float     # 平均对局时长(分钟)

    @property
    def score(self) -> float:
        """加权汇总到30分"""
        s = 0
        s += min(5, self.base_winrate * 100)                    # 基础 5%
        s += min(10, self.strategy_winrate * 100)               # 策略 25%
        s += min(5, max(0, 5 - abs(0.5 - self.floor_gradient)))# 梯度
        s += min(5, self.choice_diversity * 10)                 # 多样性
        s += min(5, max(0, 5 - abs(15 - self.game_length_min))) # 时长 15min
        return round(s, 1)

3A. 平衡性模拟

# —————————————— scripts/quality_balance.py ——————————————

class BalanceAssessor:
    """数值平衡评估器"""

    def assess_balance(self, simulation_results: dict) -> dict:
        """
        输入: simulation_results = {
            'sword': {'wins': 320, 'total': 1000, 'avg_hp': 35},
            'shield': {'wins': 290, 'total': 1000, 'avg_hp': 42},
            'spell': {'wins': 340, 'total': 1000, 'avg_hp': 30},
        }
        """
        win_rates = {k: v['wins']/v['total'] for k, v in simulation_results.items()}

        # 指标1: 最大流派差值 (10分)
        max_diff = max(win_rates.values()) - min(win_rates.values())
        balance_score = self._score_diff(max_diff)
        # <5% → 10分, 5-10% → 6分, 10-15% → 3分, >15% → 0分

        # 指标2: 卡牌选取率均匀度 (5分)
        card_pick_rates = self._analyze_card_picks()
        card_cv = self._coefficient_of_variation(list(card_pick_rates.values()))
        card_score = self._score_cv(card_cv)
        # CV<0.5 → 5分, CV<1.0 → 3分, else 0

        # 指标3: 费用效率线性度 (5分)
        cost_efficiency = self._check_cost_curve()
        # 1费:2费:3费:4费 ≈ 1.0:2.2:3.5:5.0 线性
        curve_score = self._score_linearity(cost_efficiency)

        total = balance_score + card_score + curve_score
        return {
            'max_winrate_diff': max_diff,
            'card_pick_cv': card_cv,
            'total_score': total,
            'pass': total >= 14,  # 70% 通过线
        }

    def _score_diff(self, diff: float) -> int:
        if diff < 0.05: return 10
        if diff < 0.10: return 6
        if diff < 0.15: return 3
        return 0

    def _score_cv(self, cv: float) -> int:
        if cv < 0.3: return 5
        if cv < 0.5: return 4
        if cv < 0.8: return 2
        return 0

    def _coefficient_of_variation(self, values: list) -> float:
        import statistics
        mean = statistics.mean(values)
        if mean == 0: return 1.0
        return statistics.stdev(values) / mean

# 难度曲线应符合：前3层平缓 → 4-6加速 → 7-9陡峭 → 10层BOSS
# 验证方式：对每层模拟1000场战斗，统计平均HP消耗

def validate_difficulty_curve():
    floor_stats = {}
    for floor in range(1, 11):
        hp_losses = [simulate_floor(floor) for _ in range(1000)]
        floor_stats[floor] = {
            'mean_hp_loss': statistics.mean(hp_losses),
            'p95_hp_loss': sorted(hp_losses)[950],
        }

    # 检查非线性增长
    deltas = [floor_stats[f+1]['mean_hp_loss'] - floor_stats[f]['mean_hp_loss']
              for f in range(1, 10)]
    # 单调递增
    assert all(d > 0 for d in deltas), "难度不单调递增"
    # 后半段增幅更大（t检验）
    early = sum(deltas[:5])
    late = sum(deltas[5:])
    assert late > early * 1.5, "后期难度增幅不足"

# —————————————— scripts/quality_heuristic.py ——————————————

def heuristic_checklist() -> list:
    """自动化启发式检查项"""
    checks = [
        # 1. 可访问性 (0.5分)
        ('所有UI按钮有触摸反馈', check_button_feedback),
        ('文字可读', check_font_readability),

        # 2. 防误触 (0.5分)
        ('退出游戏有确认弹窗', check_quit_confirm),
        ('误触不会执行错误操作', check_misclick_protection),

        # 3. 信息可视性 (1分)
        ('HP/AP/金币始终可见', check_hud_visibility),
        ('卡牌费用清晰标识', check_card_cost_visible),
        ('敌人意图有指示', check_enemy_intent),

        # 4. 操作可达 (1分)
        ('手指可够到棋盘所有格子', check_reachability),
        ('卡牌拖拽有吸附反馈', check_drag_snap),

        # 5. 反馈及时 (1分)
        ('出牌有动画/音效', check_card_play_feedback),
        ('受伤有屏幕震动/闪烁', check_damage_feedback),
        ('战斗结算有过渡', check_result_transition),

        # 6. 容错 (0.5分)
        ('出牌后可取消', check_cancel_card),
        ('回合结束前可撤回', check_undo_turn),

        # 7. 新手引导 (0.5分)
        ('首次游戏有教程', check_tutorial),
        ('卡牌效果可长按查看详情', check_card_detail),
    ]

    passed = 0
    results = []
    for name, fn in checks:
        try:
            result = fn()
            if result: passed += 1
            results.append({'check': name, 'pass': result})
        except Exception as e:
            results.append({'check': name, 'pass': False, 'error': str(e)})

    # 5分制
    score = round(passed / len(checks) * 5, 1)
    return {'score': score, 'passed': passed, 'total': len(checks), 'details': results}

# —————————————— scripts/quality_report.py ——————————————

def generate_full_report() -> dict:
    """生成完整质量报告"""
    report = {
        'timestamp': datetime.now().isoformat(),
        'version': get_git_version(),
        'dimensions': {},
        'total_score': 0,
    }

    # D1 可玩性 (30分)
    p = PlayabilityAssessor().assess()
    report['dimensions']['playability'] = {
        'score': p.score,
        'max': 30,
        'details': p.__dict__,
    }

    # D2 代码质量 (20分)
    c = CodeQualityAssessor().assess()
    report['dimensions']['code_quality'] = {
        'score': c.score,
        'max': 20,
        'details': c.__dict__,
    }

    # D3 数值平衡 (20分)
    b = BalanceAssessor().assess(simulate_games(1000))
    report['dimensions']['balance'] = {
        'score': b['total_score'],
        'max': 20,
        'details': b,
    }

    # D4 性能 (15分)
    perf = PerformanceAssessor().assess()
    report['dimensions']['performance'] = {
        'score': perf.score,
        'max': 15,
    }

    # D5 内容完整性 (10分)
    content = ContentCompleteness().score()
    report['dimensions']['content'] = {
        'score': content,
        'max': 10,
    }

    # D6 启发式 (5分)
    heuristic = HeuristicAssessor().score()
    report['dimensions']['heuristic'] = {
        'score': heuristic,
        'max': 5,
    }

    report['total_score'] = sum(d['score'] for d in report['dimensions'].values())
    report['pass'] = report['total_score'] >= 70

    return report

def generate_html_report(report: dict) -> str:
    """生成可视化 HTML 报告"""
    html = f"""<h1>墨境棋局 - 质量报告</h1>
    <p>版本: {report['version']} | {report['timestamp']}</p>
    <h2>总分: {report['total_score']}/100 ({'通过' if report['pass'] else '未通过'})</h2>
    <table>
    <tr><th>维度</th><th>得分</th><th>满分</th><th>状态</th></tr>"""
    for name, d in report['dimensions'].items():
        status = '✅' if d['score'] >= d['max'] * 0.7 else '⚠️'
        html += f"<tr><td>{name}</td><td>{d['score']}</td><td>{d['max']}</td><td>{status}</td></tr>"
    html += "</table>"
    return html

// 校验标准（写入 CI）：
function validateSpriteSheet(sheet: HTMLImageElement): ValidationResult {
  const W = 200, H = 200;
  if (sheet.width !== 10 * W || sheet.height !== 4 * H)
    return { pass: false, reason: `尺寸错误: ${sheet.width}x${sheet.height}` };

  // 检查每行的第1帧（站立帧）与后续帧的颜色直方图差异
  for (let row = 0; row < 4; row++) {
    const standHist = extractColorHistogram(sheet, 0, row);
    for (let col = 1; col < 10; col++) {
      const frameHist = extractColorHistogram(sheet, col, row);
      const diff = histogramDiff(standHist, frameHist);
      if (diff > 0.3) // 阈值：>30%差异则警报
        return { pass: false, reason: `Row${row} Col${col}: 风格偏差 ${diff.toFixed(2)}` };
    }
  }
  return { pass: true };
}

async function batchGenerateCards(): Promise<CardAsset[]> {
  const cards = Object.entries(cardPrompts);
  const BATCH_SIZE = 4;
  const results: CardAsset[] = [];

  for (let i = 0; i < cards.length; i += BATCH_SIZE) {
    const batch = cards.slice(i, i + BATCH_SIZE);
    // 并行发送 ImageGen 请求
    const images = await Promise.all(

      data: img,
      width: 400,
      height: 560,

  return results;

# Python: remove_background_batch.py
from PIL import Image
import numpy as np
from pathlib import Path

def make_transparent(img_path: str, output_path: str, threshold=240):
    """将接近白色的像素转为透明"""
    img = Image.open(img_path).convert("RGBA")
    arr = np.array(img)
    # 找出 R/G/B 都 > threshold 的像素
    mask = (arr[:,:,0] > threshold) & (arr[:,:,1] > threshold) & (arr[:,:,2] > threshold)
    arr[mask, 3] = 0  # 设置 alpha=0
    Image.fromarray(arr).save(output_path, "PNG")

    for (let r = 0; r < 5; r++) {
      for (let c = 0; c < 5; c++) {

        ctx.strokeRect(x + 2, y + 2, CELL - 4, CELL - 4);

        // 角落 → 浓墨 · 中心 → 淡墨

        ctx.beginPath();
        ctx.fillStyle = isCenter ? 'rgba(0,0,0,0.4)' : 'rgba(0,0,0,0.15)';
        ctx.arc(x + CELL / 2, y + CELL / 2, isCenter ? 6 : 3, 0, Math.PI * 2);
        ctx.fill();
      }
    }
    canvas.refresh();
  }
}

// —————————————— AssetLoader.ts ——————————————
// 微信小游戏 Phaser 分包加载实现

export class AssetLoader {
  private loadedPacks: Set<string> = new Set(['core']);

  /** 加载指定资源包 */
  async loadPack(packName: string, scene: Phaser.Scene): Promise<void> {
    if (this.loadedPacks.has(packName)) return;

    return new Promise((resolve, reject) => {
      // 微信小游戏分包路径规则
      const packPath = `subpacks/${packName}/assets.json`;

      scene.load.multiatlas(packName, packPath, `subpacks/${packName}`);
      scene.load.start();

      scene.load.once('complete', () => {
        this.loadedPacks.add(packName);
        resolve();
      });
      scene.load.once('loaderror', (file: any) => {
        reject(new Error(`分包加载失败: ${packName}, 文件: ${file.key}`));
      });
    });
  }

  /** 按进度预判加载（Predictive Loading）：
   *  根据当前楼层和流派，预加载下一层会用到的卡牌包 */
  predictiveLoad(floor: number, build: string[]): void {
    const nextPack = this.predictNextPack(floor, build);
    if (nextPack && !this.loadedPacks.has(nextPack)) {
      // 异步预加载，不阻塞主流程
      this.loadPack(nextPack, this.getCurrentScene()).catch(() => {});
    }
  }

  private predictNextPack(floor: number, build: string[]): string | null {
    if (floor <= 3) return 'layer1_2';
    if (floor <= 6) return 'layer4_6';
    if (floor <= 9) return 'layer7_9';
    return 'boss';
  }
}

1.7 风格一致性保障

AI 生成素材最大的风险是风格飘移。以下策略组合使用：

  1. 统一 prompt 前缀：所有素材生成时加上 CARD_STYLE 常量（水墨国风 + 黑白灰 + 毛笔笔触 + 留白），保持视觉 DNA
  2. 种子固定：ImageGen 参数传固定 seed（如 42），至少在单一角色/卡牌内部固定，减少随机变异
  3. 参考图驱动：sprite-sheet-generator 通过 read_file 加载参考图后再生图，AI 自动参考风格
  4. 后处理归一化：所有素材经过统一色调映射脚本（Python Pillow），校正色相偏移
  5. 备用方案：最终效果不满意时，使用 canvas-design 生成纯几何/矢量替代素材（低配但风格可控）
# Python: style_normalize.py — 素材风格归一化
from PIL import Image, ImageEnhance
import numpy as np
from pathlib import Path

def normalize_ink_style(img_path: str, output_path: str):
    """将素材调整到水墨风格色域"""
    img = Image.open(img_path).convert("RGBA")

    # 1. 降低饱和度（水墨风=低饱和）
    enhancer = ImageEnhance.Color(img)
    img = enhancer.enhance(0.4)

    # 2. 增加对比度（墨迹更清晰）
    enhancer = ImageEnhance.Contrast(img)
    img = enhancer.enhance(1.3)

    # 3. 色相偏移：使偏色向黑白灰收敛
    arr = np.array(img)
    gray = arr[:,:,:3].mean(axis=2, keepdims=True)  # (H,W,1)
    # 混合：70% 灰度值 + 30% 原色
    arr[:,:,:3] = (0.7 * gray + 0.3 * arr[:,:,:3]).astype(np.uint8)
    img = Image.fromarray(arr)

    img.save(output_path, "PNG")
    print(f"  ✓ {Path(img_path).name} → 风格归一化完成")

# 批量处理
for p in Path("assets/cards").glob("*.png"):
    normalize_ink_style(str(p), str(p))

2.2 资源加载队列（代码级实现）

// —————————————— QoSAssetQueue.ts ——————————————
// 精细化的资源加载队列，支持优先级、超时、重试、降级

export enum AssetPriority { CRITICAL, HIGH, NORMAL, LOW, DEFERRED }

interface QueueItem {
  key: string;
  url: string;
  priority: AssetPriority;
  retries: number;
  maxRetries: number;
  timeout: number;         // 单次超时（ms）
  fallback?: string;       // 降级资源key
  onProgress?: (pct: number) => void;
}

export class QoSAssetQueue {
  private queue: QueueItem[] = [];
  private loading = false;
  private stats = { total: 0, loaded: 0, failed: 0, retries: 0 };

  enqueue(item: QueueItem): void {
    // 按优先级插入
    const idx = this.queue.findIndex(q => q.priority < item.priority);
    if (idx === -1) this.queue.push(item);
    else this.queue.splice(idx, 0, item);
    this.stats.total++;
    this.processNext();
  }

  private async processNext(): Promise<void> {
    if (this.loading || this.queue.length === 0) return;
    this.loading = true;

    const item = this.queue.shift()!;
    const result = await this.loadWithRetry(item);

    if (!result.success) {
      this.stats.failed++;
      // 降级策略：使用 fallback 资源
      if (item.fallback) {
        console.warn(`[QoS] ${item.key} 加载失败，降级至 ${item.fallback}`);
        await this.loadDirect(item.fallback);
      } else {
        // 没有降级，用程序化生成的占位图
        this.generatePlaceholder(item.key);
        console.error(`[QoS] ${item.key} 加载失败，使用占位图`);
      }
    } else {
      this.stats.loaded++;
    }

    this.loading = false;
    this.processNext();
  }

  private async loadWithRetry(item: QueueItem, attempt = 0): Promise<{success: boolean}> {
    try {
      const controller = new AbortController();
      const timer = setTimeout(() => controller.abort(), item.timeout);

      // 模拟 Phaser.load 的 Promise 封装
      await this.loadAsset(item.key, item.url, controller.signal);
      clearTimeout(timer);
      return { success: true };
    } catch (err) {
      if (attempt < item.maxRetries) {
        this.stats.retries++;
        // 指数退避：1s, 2s, 4s...
        const delay = Math.pow(2, attempt) * 1000;
        console.warn(`[QoS] ${item.key} 重试 #${attempt + 1}, ${delay}ms后`);
        await sleep(delay);
        return this.loadWithRetry(item, attempt + 1);
      }
      return { success: false };
    }
  }

  /** 降级：程序化生成占位图 */
  private generatePlaceholder(key: string): void {
    const canvas = document.createElement('canvas');
    canvas.width = 200; canvas.height = 200;
    const ctx = canvas.getContext('2d')!;
    ctx.fillStyle = '#2a2a3e';
    ctx.fillRect(0, 0, 200, 200);
    ctx.fillStyle = '#888';
    ctx.font = '16px sans-serif';
    ctx.textAlign = 'center';
    ctx.fillText('加载中', 100, 100);
    // 注册到 Phaser 纹理缓存
    // this.scene.textures.addCanvas(key, canvas);
  }
}

2.3 帧率保障（CPU/GPU 预算控制）

// —————————————— FrameBudget.ts ——————————————
// 每帧时间预算控制器

export interface FrameBudgetReport {
  fps: number;
  used: number;     // 实际耗时(ms)
  budget: number;   // 预算(ms)
  overage: number;  // 超支(ms)
  warnings: string[];
}

export class FrameBudget {
  private static readonly TARGET_FPS = 30;
  private static readonly FRAME_BUDGET_MS = 1000 / FrameBudget.TARGET_FPS; // 33.3ms
  private static readonly WARN_THRESHOLD = 0.8; // 超过80%预算即警告

  private history: number[] = [];
  private lastFrameTime = 0;

  /** 每帧开始时调用，返回剩余时间预算 */
  beginFrame(): number {
    const now = performance.now();
    if (this.lastFrameTime > 0) {
      const elapsed = now - this.lastFrameTime;
      this.history.push(elapsed);
      if (this.history.length > 60) this.history.shift(); // 保留最近60帧
    }
    this.lastFrameTime = now;
    return FrameBudget.FRAME_BUDGET_MS;
  }

  /** 每帧结束时报告 */
  endFrame(): FrameBudgetReport {
    const elapsed = performance.now() - this.lastFrameTime;
    const fps = this.calcFPS();
    const overage = Math.max(0, elapsed - FrameBudget.FRAME_BUDGET_MS);

    const warnings: string[] = [];
    const usage = elapsed / FrameBudget.FRAME_BUDGET_MS;
    if (usage > FrameBudget.WARN_THRESHOLD) {
      warnings.push(`帧预算使用率 ${(usage * 100).toFixed(0)}%，剩余 ${(FrameBudget.FRAME_BUDGET_MS - elapsed).toFixed(1)}ms`);
    }
    if (fps < 20) {
      warnings.push(`⚠️ 严重掉帧: ${fps.toFixed(1)} FPS`);
    }

    return { fps, used: elapsed, budget: FrameBudget.FRAME_BUDGET_MS, overage, warnings };
  }

  private calcFPS(): number {
    if (this.history.length < 2) return FrameBudget.TARGET_FPS;
    const avg = this.history.reduce((a, b) => a + b, 0) / this.history.length;
    return avg > 0 ? 1000 / avg : FrameBudget.TARGET_FPS;
  }

  /** 动态质量缩放：检测低帧率时自动降质 */
  static autoQualityScale(fps: number): number {
    if (fps >= 30) return 1.0; // 高品质
    if (fps >= 25) return 0.75; // 中度降质：关部分特效
    if (fps >= 20) return 0.5;  // 大幅度降质：关粒子系统
    return 0.25;                 // 最低质量：所有特效关闭，简化渲染
  }
}

3.2 错误边界与恢复

// —————————————— core/ErrorBoundary.ts ——————————————
// 错误边界：防止单点崩溃导致整个游戏不可用

type ErrorHandler = (error: Error, context: string) => void;

export class ErrorBoundary {
  private static handlers: ErrorHandler[] = [];

  static registerHandler(handler: ErrorHandler): void {
    this.handlers.push(handler);
  }

  /** 包裹一个函数，捕获内部异常并安全降级 */
  static wrap<T>(fn: () => T, context: string, fallback: T): T {
    try {
      return fn();
    } catch (err) {
      const error = err instanceof Error ? err : new Error(String(err));
      this.handlers.forEach(h => h(error, context));
      console.error(`[ErrorBoundary] ${context}:`, error);
      return fallback;
    }
  }

  /** 异步版本 */
  static async wrapAsync<T>(
    fn: () => Promise<T>, context: string, fallback: T
  ): Promise<T> {
    try {
      return await fn();
    } catch (err) {
      const error = err instanceof Error ? err : new Error(String(err));
      this.handlers.forEach(h => h(error, context));
      console.error(`[ErrorBoundary] ${context}:`, error);
      return fallback;
    }
  }

  /** 装饰器风格（对对象的方法批量包裹） */
  static protectMethods<T extends object>(
    obj: T, contexts: Record<string, string>, fallbacks: Record<string, any>
  ): T {
    const proxy = new Proxy(obj, {
      get(target, prop: string) {
        const orig = (target as any)[prop];
        if (typeof orig !== 'function') return orig;
        const ctx = contexts[prop] || prop;
        const fb = fallbacks[prop];
        return (...args: any[]) => {
          try { return orig.apply(target, args); }
          catch (err) {
            console.error(`[ErrorBoundary] ${ctx}:`, err);
            return fb;
          }
        };
      }
    });
    return proxy;
  }
}

// —————————————— core/GameStateMachine.ts ——————————————
// 严格状态机确保游戏流程不会"走进死胡同"

export enum GameState {
  BOOT       = 'BOOT',
  MENU       = 'MENU',
  INIT_RUN   = 'INIT_RUN',
  PLAYER_TURN= 'PLAYER_TURN',
  ENEMY_TURN = 'ENEMY_TURN',
  RESOLVING  = 'RESOLVING',
  REWARD     = 'REWARD',
  GAME_OVER  = 'GAME_OVER',
  PAUSED     = 'PAUSED',
}

const TRANSITIONS: Record<GameState, GameState[]> = {
  [GameState.BOOT]:        [GameState.MENU],
  [GameState.MENU]:        [GameState.INIT_RUN],
  [GameState.INIT_RUN]:    [GameState.PLAYER_TURN],
  [GameState.PLAYER_TURN]: [GameState.ENEMY_TURN, GameState.PAUSED],
  [GameState.ENEMY_TURN]:  [GameState.RESOLVING],
  [GameState.RESOLVING]:   [GameState.PLAYER_TURN, GameState.REWARD, GameState.GAME_OVER],
  [GameState.REWARD]:      [GameState.PLAYER_TURN, GameState.GAME_OVER],
  [GameState.GAME_OVER]:   [GameState.MENU],
  [GameState.PAUSED]:      [GameState.PLAYER_TURN, GameState.MENU],
};

export class GameStateMachine {
  private current: GameState = GameState.BOOT;
  private history: GameState[] = [];

  get state(): GameState { return this.current; }

  transition(to: GameState): boolean {
    const allowed = TRANSITIONS[this.current];
    if (!allowed || !allowed.includes(to)) {
      console.error(
        `[StateMachine] 非法转移: ${this.current} → ${to}. ` +
        `允许: [${allowed?.join(', ') || '无'}]`
      );
      return false; // 安全拒绝非法转移
    }
    this.history.push(this.current);
    this.current = to;
    console.log(`[StateMachine] ${this.history[this.history.length - 1]} → ${to}`);
    return true;
  }

  /** 回滚到上一个状态（错误恢复用） */
  rollback(): boolean {
    if (this.history.length === 0) return false;
    this.current = this.history.pop()!;
    console.log(`[StateMachine] 回滚至: ${this.current}`);
    return true;
  }

  /** 快照：用于异常时的状态恢复 */
  snapshot(): string {
    return JSON.stringify({
      state: this.current,
      history: this.history,
    });
  }

  restore(snapshot: string): void {
    const data = JSON.parse(snapshot);
    this.current = data.state;
    this.history = data.history;
  }
}

// —————————————— data/SaveManager.ts ——————————————

interface SaveData {
  version: number;         // 存档版本号，用于向前兼容
  checksum: string;        // SHA-256 校验和
  timestamp: number;
  player: {
    hp: number; maxHp: number;
    atk: number; arm: number;
    deck: string[];        // 卡牌ID列表
    gold: number;
  };
  progress: {
    floor: number;
    clearedRooms: number[];
    relics: string[];
  };
  meta: {
    totalRuns: number;
    totalWins: number;
  };
}

export class SaveManager {
  private static readonly SAVE_KEY = 'ink_realm_save';
  private static readonly SAVE_VERSION = 1;

  /** 带校验的保存 */
  static save(data: SaveData): boolean {
    try {
      data.version = SaveManager.SAVE_VERSION;
      data.timestamp = Date.now();
      data.checksum = SaveManager.calcChecksum(data);
      wx.setStorageSync(SaveManager.SAVE_KEY, JSON.stringify(data));
      return true;
    } catch (err) {
      console.error('[SaveManager] 保存失败:', err);
      // 微信存储满时会抛出异常
      return false;
    }
  }

  /** 带完整性校验的读取 */
  static load(): SaveData | null {
    try {
      const raw = wx.getStorageSync(SaveManager.SAVE_KEY);
      if (!raw) return null;

      const data: SaveData = JSON.parse(raw);
      const expectedChecksum = data.checksum;

      // 版本迁移
      const migrated = SaveManager.migrate(data);
      if (!migrated) {
        console.error('[SaveManager] 版本迁移失败');
        return null;
      }

      // 校验和验证
      const actualChecksum = SaveManager.calcChecksum(migrated);
      if (actualChecksum !== expectedChecksum) {
        console.error('[SaveManager] 存档校验失败，数据可能损坏');
        return null; // 返回 null → 启动新游戏
      }

      // 数据合理性校验
      if (!SaveManager.validate(migrated)) {
        console.error('[SaveManager] 存档数据不合理');
        return null;
      }

      return migrated;
    } catch (err) {
      console.error('[SaveManager] 读取失败:', err);
      return null;
    }
  }

  private static calcChecksum(data: SaveData): string {
    const { checksum, ...rest } = data;
    // 微信小游戏下使用 SHA-256，简化版用 hashCode
    // 实际应使用 wx.getFileSystemManager 或 crypto.subtle
    return simpleHash(JSON.stringify(rest));
  }

  private static validate(data: SaveData): boolean {
    // 血量不超上限
    if (data.player.hp > data.player.maxHp) return false;
    // 卡组不超上限
    if (data.player.deck.length > 30) return false;
    // 楼层范围合理
    if (data.progress.floor < 0 || data.progress.floor > 10) return false;
    return true;
  }

  private static migrate(data: any): SaveData | null {
    while (data.version < SaveManager.SAVE_VERSION) {
      if (data.version === 0) {
        // v0 → v1: 增加 checksum 字段
        // 之前的存档无 checksum，自动生成
      }
      data.version++;
    }
    return data as SaveData;
  }
}

// —————————————— debug/Logger.ts ——————————————
// 分级日志系统：开发环境全量输出，生产环境只保留 ERROR + 采样 WARN

export enum LogLevel {
  DEBUG = 0, INFO = 1, WARN = 2, ERROR = 3, SILENT = 4,
}

interface LogEntry {
  timestamp: number;
  level: LogLevel;
  module: string;
  message: string;
  data?: any;
  stack?: string;
}

export class Logger {
  private static level: LogLevel = isDev() ? LogLevel.DEBUG : LogLevel.ERROR;
  private static buffer: LogEntry[] = [];
  private static readonly MAX_BUFFER = 200;
  private static readonly FLUSH_INTERVAL = 30000; // 30s 刷一次

  static init(): void {
    // 生产环境 WARN 级别按 10% 采样
    if (!isDev()) {
      Logger.level = LogLevel.ERROR;
    }
    // 定时刷日志到远程
    setInterval(() => Logger.flush(), Logger.FLUSH_INTERVAL);
  }

  static debug(module: string, message: string, data?: any): void {
    if (Logger.level > LogLevel.DEBUG) return;
    Logger.push(LogLevel.DEBUG, module, message, data);
  }

  static info(module: string, message: string, data?: any): void {
    if (Logger.level > LogLevel.INFO) return;
    Logger.push(LogLevel.INFO, module, message, data);
  }

  static warn(module: string, message: string, data?: any): void {
    if (Logger.level > LogLevel.WARN) return;
    // 生产环境 10% 采样
    if (!isDev() && Math.random() > 0.1) return;
    Logger.push(LogLevel.WARN, module, message, data);
  }

  static error(module: string, message: string, data?: any): void {
    if (Logger.level > LogLevel.ERROR) return;
    const entry = Logger.push(LogLevel.ERROR, module, message, data);
    // ERROR 立即上报
    Logger.reportToRemote(entry);
  }

  private static push(level: LogLevel, module: string, message: string, data?: any): LogEntry {
    const entry: LogEntry = {
      timestamp: Date.now(),
      level,
      module,
      message,
      data: data ? Logger.sanitize(data) : undefined,
      stack: level >= LogLevel.WARN ? new Error().stack?.split('\n').slice(2, 6).join('\n') : undefined,
    };
    Logger.buffer.push(entry);
    if (Logger.buffer.length > Logger.MAX_BUFFER) Logger.buffer.shift();

    // 控制台输出
    const prefix = `[${LogLevel[level]}] [${module}]`;
    switch (level) {
      case LogLevel.DEBUG: console.debug(prefix, message, data ?? ''); break;
      case LogLevel.INFO:  console.info(prefix, message, data ?? ''); break;
      case LogLevel.WARN:  console.warn(prefix, message, data ?? ''); break;
      case LogLevel.ERROR: console.error(prefix, message, data ?? ''); break;
    }

    return entry;
  }

  /** 敏感信息脱敏 */
  private static sanitize(data: any): any {
    const str = JSON.stringify(data);
    // 脱敏 token, key, session 等字段
    return JSON.parse(str.replace(/(sk-|session_key|token")[^"]+/g, '$1***'));
  }

  /** 上报到远程日志服务 */
  private static reportToRemote(entry: LogEntry): void {
    // 生产环境通过 wx.request 上报
    if (!isDev()) {
      wx.request({
        url: 'https://log.inkrealm.com/report',
        method: 'POST',
        data: { entries: [entry] },
        fail: () => {} // 静默失败，不影响游戏
      });
    }
  }

  /** 获取最近日志用于调试面板 */
  static getRecentLogs(count: number = 50): LogEntry[] {
    return Logger.buffer.slice(-count);
  }

  static flush(): void {
    if (Logger.buffer.length === 0) return;
    const batch = Logger.buffer.splice(0, Logger.MAX_BUFFER);
    if (!isDev() && batch.some(e => e.level === LogLevel.ERROR)) {
      wx.request({
        url: 'https://log.inkrealm.com/report',
        method: 'POST',
        data: { entries: batch.slice(0, 20) }, // 最多 20 条
        fail: () => {}
      });
    }
  }
}

    // 使用 DOM 创建调试面板（Phaser 不支持复杂UI，微信小游戏可用 DOM）
    const panel = document.createElement('div');
    panel.id = 'dev-panel';
    panel.style.cssText = `
      position: fixed; top: 0; left: 0; width: 100%; height: 100%;
      background: rgba(0,0,0,0.85); z-index: 9999;
      display: flex; flex-direction: column; padding: 20px;
      font-size: 12px; color: #0f0; font-family: monospace;
      overflow-y: auto;
    `;
    panel.innerHTML = `
      
⚙ 墨境棋局 Debug 面板

      

        
        
        
        
        
      

      

      
    `;
    document.body.appendChild(panel);

    // Tab 切换
    panel.querySelectorAll('[data-tab]').forEach(btn => {
      btn.addEventListener('click', () => {
        const tab = (btn as HTMLElement).dataset.tab!;
        DevPanel.renderTab(tab);
      });
    });
    panel.querySelector('#dev-panel-close')!.addEventListener('click', () => {
      DevPanel.hide(scene);
    });

    // 默认显示日志
    DevPanel.renderTab('log');
  }

  private static renderTab(tab: string): void {
    const content = document.getElementById('dev-panel-content')!;
    switch (tab) {
      case 'log': {
        const logs = Logger.getRecentLogs(100);
        content.innerHTML = logs.map(l =>
          `

            [${new Date(l.timestamp).toISOString().slice(11,19)}] [${LogLevel[l.level]}] [${l.module}] ${l.message}
          
`
        ).join('');
        break;
      }
      case 'state': {
        const fsm = (window as any).gameStateMachine;
        content.innerHTML = `
          
当前状态: ${fsm?.state ?? 'N/A'}

          
玩家HP: ${fsm?.player?.hp}/${fsm?.player?.maxHp}

          
楼层: ${fsm?.floor ?? 'N/A'}

          
卡组: ${(fsm?.player?.deck?.length ?? 0)} 张

        `;
        break;
      }
      case 'perf': {
        content.innerHTML = '
FPS: --
';
        setInterval(() => {
          const perf = wx.getPerformance();
          document.getElementById('fps-counter')!.textContent =
            `${perf.fps?.toFixed(1) ?? 'N/A'} fps`;
        }, 1000);
        break;
      }
      case 'memory': {
        const mem = wx.getPerformance().memory;
        content.innerHTML = `
          
JS堆: ${(mem?.used / 1024 / 1024).toFixed(1)} MB / ${(mem?.limit / 1024 / 1024).toFixed(1)} MB

          
纹理缓存: ${Object.keys(game.textures?.keys ?? {}).length} 个

        `;
        break;
      }
      case 'network': {
        content.innerHTML = `
          
资源加载队列: ${QoSAssetQueue.stats.loaded} / ${QoSAssetQueue.stats.total}

          
重试次数: ${QoSAssetQueue.stats.retries}

          
失败次数: ${QoSAssetQueue.stats.failed}

        `;
        break;
      }
    }
  }

  private static hide(scene: Phaser.Scene): void {
    document.getElementById('dev-panel')?.remove();
  }
}

4.4 远程日志上报

// —————————————— debug/RemoteLogger.ts ——————————————

export class RemoteLogger {
  private static readonly ENDPOINT = 'https://log.inkrealm.com/report';
  private static sessionId = '';

  static init(): void {
    RemoteLogger.sessionId = `session_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`;
  }

  /** 错误上报 */
  static error(module: string, message: string, stack?: string): void {
    const payload = {
      sessionId: RemoteLogger.sessionId,
      type: 'ERROR',
      module,
      message,
      stack,
      timestamp: Date.now(),
      network: wx.getNetworkType?.() ?? 'unknown',
      device: wx.getSystemInfoSync?.().model ?? 'unknown',
      version: '1.0.0', // 游戏版本号
    };

    // 使用 Beacon API 避免影响主流程
    wx.request({
      url: RemoteLogger.ENDPOINT,
      method: 'POST',
      data: payload,
      fail: () => {}, // 静默
    });
  }

  /** 自定义埋点（用于监控关键路径） */
  static track(event: string, data?: Record<string, any>): void {
    if (!isDev()) {
      wx.request({
        url: RemoteLogger.ENDPOINT,
        method: 'POST',
        data: {
          sessionId: RemoteLogger.sessionId,
          type: 'TRACK',
          event,
          data,
          timestamp: Date.now(),
        },
        fail: () => {},
      });
    }
  }
}

// —————————————— debug/DebugBoot.ts ——————————————
// 统一调试初始化入口

export class DebugBoot {
  static async init(scene: Phaser.Scene): Promise<void> {
    // 1. 日志系统
    Logger.init();

    // 2. 全局错误捕获
    WxStability.init();

    // 3. 远程日志 session
    RemoteLogger.init();

    // 4. 性能监控（生产环境采样 1%）
    if (isDev() || Math.random() < 0.01) {
      PerfMonitor.start();
    }

    // 5. 调试面板（仅开发/体验版）
    if (isDev() || __wxConfig?.envVersion === 'trial') {
      DevPanel.init(scene);

      // 注册全局快捷键
      document.addEventListener('keydown', (e) => {
        if (e.key === 'F12') DevPanel.toggle(scene);
      });
    }

    // 6. 上报启动事件
    RemoteLogger.track('game_start', {
      env: __wxConfig?.envVersion ?? 'develop',
      device: wx.getSystemInfoSync().model,
      sdkVersion: wx.getSystemInfoSync().SDKVersion,
    });

    Logger.info('DebugBoot', '调试系统初始化完成', {
      dev: isDev(),
      fps: 30,
      logLevel: LogLevel[Logger.level],
    });
  }
}

// —————————————— 辅助函数 ——————————————
function isDev(): boolean {
  // 通过 wx 环境变量判断
  try {
    const account = wx.getAccountInfoSync();
    return account.miniProgram.envVersion === 'develop';
  } catch {
    return true; // 非微信环境视为开发环境
  }
}

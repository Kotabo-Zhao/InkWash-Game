地图生成算法

每次开局生成一张 4×6 节点地图（4行×6列），玩家从底部起点出发，逐层选择前进路线到达顶部 BOSS。

// —————————————— map/MapGenerator.ts ——————————————

export enum NodeType {
  BATTLE = 'BATTLE',       // 普通战斗
  ELITE = 'ELITE',         // 精英战斗 → 高奖励
  BOSS = 'BOSS',           // BOSS战
  REST = 'REST',           // 休息（回血/升级）
  SHOP = 'SHOP',           // 商店
  EVENT = 'EVENT',         // 随机事件
  TREASURE = 'TREASURE',   // 宝箱（免费卡牌）
  MYSTERY = 'MYSTERY',     // 未知（进入前不可见类型）
}

interface MapNode {
  id: string;
  type: NodeType;
  row: number;            // 0=BOSS层, 5=起点层
  col: number;
  connections: string[];  // 连接的下层节点ID
  isRevealed: boolean;    // 是否已探索
}

export class MapGenerator {
  static readonly ROWS = 6;
  static readonly COLS = 6;

  /** 生成一张随机地图 */
  static generate(seed: number): MapNode[][] {
    // 种子随机数确保同一 seed 生成相同地图
    const rng = new SeededRandom(seed);

    // 第0行：BOSS 固定
    // 第1-4行：随机分布
    // 第5行：起点（固定）

    const grid: MapNode[][] = [];

    // 第5行（起点）— 固定第3列为起点
    grid[5] = this.generateRow(5, [NodeType.REST], rng);
    grid[5][2] = { id: 'start', type: NodeType.REST, row: 5, col: 2, connections: [], isRevealed: true };

    // 第4-1行（中间4层）
    for (let row = 4; row >= 1; row--) {
      grid[row] = this.generateRow(row, this.pickRowTypes(row, rng), rng);
    }

    // 第0行（BOSS）
    grid[0] = this.generateRow(0, [NodeType.BOSS, NodeType.ELITE, NodeType.BOSS], rng);

    // 建立连接：每层节点连接下一层的2-3个节点
    this.buildConnections(grid, rng);

    return grid;
  }

  private static pickRowTypes(row: number, rng: SeededRandom): NodeType[] {
    // 不同层有不同的节点类型概率分布
    const pool: NodeType[] = [];
    const add = (type: NodeType, weight: number) => {
      for (let i = 0; i < weight; i++) pool.push(type);
    };

    // 越靠近BOSS层，精英和商店越多
    if (row >= 3) {
      add(NodeType.BATTLE, 4);
      add(NodeType.ELITE, 0);
      add(NodeType.SHOP, 1);
      add(NodeType.EVENT, 2);
      add(NodeType.REST, 1);
      add(NodeType.TREASURE, 1);
      add(NodeType.MYSTERY, 1);
    } else {
      add(NodeType.BATTLE, 3);
      add(NodeType.ELITE, 1);
      add(NodeType.SHOP, 1);
      add(NodeType.EVENT, 1);
      add(NodeType.REST, 1);
      add(NodeType.TREASURE, 1);
      add(NodeType.MYSTERY, 2);
    }

    // 洗牌后取 COLS 个
    return rng.shuffle(pool).slice(0, MapGenerator.COLS);
  }

  private static buildConnections(grid: MapNode[][], rng: SeededRandom): void {
    for (let row = 1; row < MapGenerator.ROWS; row++) {
      for (let c = 0; c < MapGenerator.COLS; c++) {
        const node = grid[row][c];
        if (!node) continue;

        // 每个节点连接下一层的 [c-1, c, c+1] 范围内的2-3个节点
        const candidates = [c];
        if (c > 0) candidates.push(c - 1);
        if (c < MapGenerator.COLS - 1) candidates.push(c + 1);

        // 取2-3个连接
        const count = rng.next() > 0.5 ? 3 : 2;
        const selected = rng.shuffle(candidates).slice(0, count);

        node.connections = selected.map(col =>
          grid[row - 1][col]?.id
        ).filter(Boolean);
      }
    }
  }

  private static generateRow(row: number, types: NodeType[], rng: SeededRandom): MapNode[] {
    const typesCopy = rng.shuffle([...types]);
    return typesCopy.map((type, col) => ({
      id: `n${row}_${col}`,
      type,
      row,
      col,
      connections: [],
      isRevealed: col === 2 && row === 5, // 仅起点可见
    }));
  }
}

// —————————————— shop/ShopSystem.ts ——————————————

export class ShopSystem {
  /** 生成商店库存 */
  static generateInventory(floor: number, rng: SeededRandom): ShopInventory {
    // 卡牌：根据楼层决定稀有度分布
    const cardCount = 4 + Math.floor(floor / 3); // 4~7张
    const cards: ShopCard[] = [];

    for (let i = 0; i < cardCount; i++) {
      const rarity = this.rollRarity(floor, rng);
      cards.push({
        cardId: this.randomCard(rarity, rng),
        price: this.cardPrice(rarity, floor),
      });
    }

    // 墨宝：1-2个
    const relicCount = 1 + (rng.next() > 0.6 ? 1 : 0);
    const relics: ShopRelic[] = [];
    for (let i = 0; i < relicCount; i++) {
      relics.push({
        relicId: this.randomRelic(rng),
        price: 120 + Math.floor(floor * 10),
      });
    }

    // 删卡服务
    const removeCost = 50 + floor * 10; // 每层涨价10

    // 回复药水
    const potion = {
      healPercent: 20 + floor * 3, // 20%~50%
      price: 40 + floor * 5,
    };

    return { cards, relics, removeCost, potion };
  }

  private static rollRarity(floor: number, rng: SeededRandom): Rarity {
    const roll = rng.next();
    // 低楼层：普通居多；高楼层：稀有/史诗概率提升
    if (floor <= 3) {
      if (roll < 0.6) return Rarity.COMMON;
      if (roll < 0.85) return Rarity.RARE;
      return Rarity.EPIC;
    } else {
      if (roll < 0.3) return Rarity.COMMON;
      if (roll < 0.6) return Rarity.RARE;
      if (roll < 0.85) return Rarity.EPIC;
      return Rarity.LEGENDARY;
    }
  }

  private static cardPrice(rarity: Rarity, floor: number): number {
    const base = { COMMON: 50, RARE: 80, EPIC: 130, LEGENDARY: 200 };
    return base[rarity] + floor * 5;
  }
}

1.5 数值公式体系

// —————————————— combat/BattleFormulas.ts ——————————————

export class BattleFormulas {
  /** ═══ 基础伤害公式 ═══
   *  最终伤害 = (基础攻击力 × 技能倍率 + 额外伤害) × (1 + 暴击加成) × 属性克制 − 护甲
   *  最少 1 点伤害 */
  static calcDamage(
    baseAtk: number, skillDamage: number,
    target: Enemy, critRate: number, elementBonus: number
  ): number {
    const isCrit = Math.random() < critRate;
    const critMult = isCrit ? 1.5 : 1.0;
    const raw = (baseAtk + skillDamage) * critMult * (1 + elementBonus);
    const final = Math.max(1, Math.floor(raw - target.armor));
    return final;
  }

  /** ═══ 暴击率公式 ═══
   *  基础 5% + 每点灵韵 2% + 卡牌/墨宝加成
   *  上限 80% */
  static calcCritRate(baseCrit: number, lingyun: number, bonuses: number): number {
    return Math.min(0.8, baseCrit + lingyun * 0.02 + bonuses);
  }

  /** ═══ 护甲减伤公式 ═══
   *  护甲值为绝对减伤（1护甲 = 1伤害减免）
   *  回合结束时护甲归零 */
  static applyArmor(damage: number, armor: number): { finalDamage: number; armorRemaining: number } {
    const absorbed = Math.min(armor, damage);
    return {
      finalDamage: damage - absorbed,
      armorRemaining: armor - absorbed,
    };
  }

  /** ═══ 费用效率（卡牌平衡基准） ═══
   *  1费=1.0x, 2费=2.2x, 3费=3.5x, 4费=5.0x, 0费=0.6x
   *  用于标准化所有卡牌的数值 */
  static costEfficiency(cost: number): number {
    const table: Record = { 0: 0.6, 1: 1.0, 2: 2.2, 3: 3.5, 4: 5.0 };
    return table[cost] ?? 1.0;
  }

  /** ═══ 敌人难度曲线 ═══
   *  楼层 → 基础属性映射 */
  static enemyStats(floor: number, isElite: boolean, isBoss: boolean): EnemyTemplate {
    const mult = isBoss ? 2.5 : isElite ? 1.6 : 1.0;

    // 非线性曲线：前3层平缓，4-6加速，7-9陡峭
    const expCurve = (f: number) => 1 + 0.15 * f + 0.02 * f * f;

    return {
      maxHp: Math.floor(20 * expCurve(floor) * mult),
      atk: Math.floor(5 * expCurve(floor) * mult * 0.6),
      armor: Math.floor(3 * expCurve(floor) * (isBoss ? 0.3 : 0)),
      // ...
    };
  }

  /** ═══ 战斗价值标准（参考值） ═══
   *  1张1费普通卡 = 6点伤害 or 5点护甲
   *  1 HP ≈ 0.8 金
   *  1 张卡牌价值 ≈ 50 金（普通）
   *  删卡价值 = 75 金 */
  static readonly STANDARDS = {
    damagePerCost: 6,      // 1费标准伤害
    blockPerCost: 5,       // 1费标准护甲
    goldPerHp: 0.8,
    goldPerCardCommon: 50,
    goldPerCardRare: 80,
    goldPerRemove: 75,
  };
}

// —————————————— combat/BattleEngine.ts ——————————————

export class BattleEngine {
  /** 执行一次完整的回合（玩家行动→敌人行动→回合结束结算） */
  async resolveTurn(turn: Turn): Promise<TurnResult> {
    Logger.info('Battle', `=== 回合开始: 玩家AP=${turn.playerAp} ===`);

    // Phase 1: 玩家行动阶段（外部事件驱动：玩家拖拽出牌）
    // 由 UI 层调用 playCard()，每打一张卡触发一次效果
    // 玩家可以消耗所有行动力，也可以提前结束回合

    // Phase 2: 敌人行动阶段
    for (const enemy of turn.enemies) {
      if (enemy.isDead) continue;

      // 冻结检测
      if (enemy.status.freeze > 0) {
        enemy.status.freeze--;
        Logger.info('Battle', `${enemy.name} 被冻结，跳过行动`);
        continue;
      }

      // 敌人 AI
      const action = this.enemyAI(enemy, turn.player);
      if (action.type === 'ATTACK') {
        const dmg = this.calcEnemyDamage(enemy, action.value);
        const { finalDamage } = BattleFormulas.applyArmor(dmg, turn.player.armor);
        turn.player.hp -= finalDamage;

        // 荆棘反伤
        if (turn.player.status.thorn > 0) {
          const thornDmg = turn.player.status.thorn * 3;
          enemy.hp -= thornDmg;
          Logger.info('Battle', `荆棘反伤 ${thornDmg} 给 ${enemy.name}`);
        }
      }
    }

    // Phase 3: 回合结束结算
    // 灼烧伤害
    for (const enemy of turn.enemies) {
      if (enemy.status.burn > 0) {
        enemy.hp -= enemy.status.burn * 2;
      }
    }

    // 护甲清零
    turn.player.armor = 0;

    // 抽牌（新回合）
    const cardsDrawn = turn.player.drawCards(4);

    Logger.info('Battle', `=== 回合结束: 玩家HP=${turn.player.hp}/${turn.player.maxHp} ===`);
    return { player: turn.player, enemies: turn.enemies, cardsDrawn };
  }

  /** 敌人 AI — 根据敌人类型选择行动 */
  private enemyAI(enemy: Enemy, player: Player): EnemyAction {
    // 简单 AI：优先攻击低血量玩家
    if (player.hp < player.maxHp * 0.3 && Math.random() < 0.6) {
      return { type: 'ATTACK', value: enemy.atk * 1.5 }; // 斩杀
    }
    if (enemy.status.armor_break > 0) {
      return { type: 'ATTACK', value: enemy.atk * 1.3 }; // 破甲后追击
    }
    if (enemy.hp < enemy.maxHp * 0.3 && Math.random() < 0.4) {
      return { type: 'BUFF', value: 3 }; // 残血狂暴
    }
    return { type: 'ATTACK', value: enemy.atk };
  }
}

# 用蒙特卡洛模拟验证数值平衡

import random
import json
from dataclasses import dataclass
from typing import List

@dataclass
class SimConfig:
    runs: int = 1000
    seed: int = 42
    strategies: List[str] = None  # ['sword', 'shield', 'spell', 'random']

def simulate_run(strategy: str, rng: random.Random) -> dict:
    """模拟一局游戏的简化版本"""
    hp = 60
    max_hp = 60
    atk = 8
    floor = 1
    deck = ['basic_strike'] * 5 + ['basic_defend'] * 5  # 起始10张

    while floor <= 10 and hp > 0:
        # 战斗简化模型
        enemy_hp = 20 + 15 * floor
        enemy_atk = 5 + 3 * floor

        # 玩家出牌循环（简化：体力全用完）
        for turn in range(10):
            if enemy_hp <= 0:
                break
            # 根据策略选择行动
            if strategy == 'sword':
                dmg = 8 + atk * 0.5 * floor
                enemy_hp -= dmg
            elif strategy == 'shield':
                # 防守反击
                dmg = 4 + atk * 0.3 * floor
                enemy_hp -= dmg
                hp = min(max_hp, hp + 2)
            elif strategy == 'spell':
                dmg = 6 * (1 + 0.2 * floor)
                enemy_hp -= dmg
                # AOE 优势：敌人多的场合更强，这里简化
            else:  # random
                dmg = 6 + random.randint(-2, 4)
                enemy_hp -= dmg

            # 敌人攻击
            hp -= max(0, enemy_atk - 3)

        if hp <= 0:
            return {'won': False, 'floor': floor, 'hp_remain': 0}

        # 升级奖励
        hp = min(max_hp, hp + 10)
        floor += 1

    return {'won': hp > 0, 'floor': floor - 1, 'hp_remain': max(0, hp)}

def run_balance_test():
    config = SimConfig(runs=5000)
    results = {s: {'wins': 0, 'total': 0} for s in ['sword', 'shield', 'spell', 'random']}

    for strategy in results:
        rng = random.Random(config.seed)
        for _ in range(config.runs):
            result = simulate_run(strategy, rng)
            results[strategy]['total'] += 1
            if result['won']:
                results[strategy]['wins'] += 1

    print(f"{'='*50}")
    print(f"平衡性模拟结果（每策略 {config.runs} 局）")
    print(f"{'='*50}")
    for strategy, data in results.items():
        win_rate = data['wins'] / data['total'] * 100
        print(f"  {strategy:>8}: {win_rate:5.1f}% 通关率")

    # 判定：最大差值 < 8% 为通过
    rates = [d['wins']/d['total'] for d in results.values()]
    max_diff = max(rates) - min(rates)
    print(f"\n  最大差值: {max_diff*100:.1f}%")
    print(f"  结果: {'✅ 通过' if max_diff < 0.08 else '❌ 需调整'}")

if __name__ == '__main__':
    run_balance_test()

二、世界观与沉浸感设计

2.1 核心设定：墨境世界

「天地初分，清气上升为天，浊气下沉为地。有一位画圣以天地为纸、山河为墨，绘出了这方世界——墨境。」

墨境（Ink Realm）是一个由「墨力」构成的世界。万物皆由墨意凝结而成：山是泼墨的堆叠，水是淡墨的流淌，生灵是点墨的活化。

千年前，画圣以《万灵图》封印了九大墨兽，维持着墨境的平衡。但如今封印松动，墨兽的气息开始侵染这片土地。玩家扮演的「墨修者」，是画圣留下的最后血脉，必须深入墨境九层，重绘封印。

2.2 世界观与玩法融合

游戏机制世界观解释沉浸感设计
卡牌 每一张卡牌是一道「墨符」——画圣流传的笔墨法术 卡牌插画是水墨风格的咒符图案；出牌时伴有墨迹飞溅粒子
卡组构筑 墨修者通过研习不同流派的墨符，形成自己的「墨道」 流派切换时角色外观变化（墨色缭绕效果不同）
HP 「墨魂」——墨修者的生命本源，墨力耗竭则消散 血条设计为墨砚中的墨水量，低血量时角色变为淡墨色
护甲 「墨甲」——以墨力凝结的防护层 护盾显示为角色周围旋转的墨色符文字环
金币 「墨玉」——墨境中流通的货币，由凝聚的墨力结晶制成 拾取动画为墨绿色晶石飞入口袋
墨宝（遗物） 上古墨修者遗留的法器，蕴含残缺的墨道真意 每个墨宝有独立的水墨典故描述
敌人 被墨兽气息污染的墨境生灵——墨化妖兽 敌人设计为传统水墨画中的异兽：墨蛟、墨虎、墨蛛
BOSS 封印中的九大墨兽之一的残影 每层BOSS都有独立出场动画和背景故事
随机事件 墨境中残存的上古记忆片段 事件以水墨卷轴动画展开，配古文风文案
路线选择 画圣留下的九层墨境塔。每层有不同路径通往深处 地图渲染为水墨山水长卷，节点为墨点标记

3.6 逐像素后处理：像素化 + 水墨化

# —————————————— scripts/pixel_ink_postprocess.py ——————————————
# 对 AI 生成的图片做后处理，确保「像素水墨风」一致性

from PIL import Image, ImageEnhance, ImageFilter
import numpy as np
from pathlib import Path

def pixelize(img: Image.Image, pixel_size: int = 8) -> Image.Image:
    """将图像像素化（降采样 + 最近邻放大）"""
    w, h = img.size
    small = img.resize((w // pixel_size, h // pixel_size), Image.NEAREST)
    return small.resize((w, h), Image.NEAREST)

def ink_wash_effect(img: Image.Image) -> Image.Image:
    """模拟水墨效果：去饱和 + 高对比 + 墨迹边缘"""
    img = img.convert("RGBA")

    # 1. 去饱和（水墨风格 = 低饱和高对比）
    arr = np.array(img)
    gray = arr[:,:,:3].mean(axis=2, keepdims=True)
    arr[:,:,:3] = (0.75 * gray + 0.25 * arr[:,:,:3]).astype(np.uint8)

    # 2. 对比度增强
    enhancer = ImageEnhance.Contrast(Image.fromarray(arr))
    img = enhancer.enhance(1.5)

    # 3. 墨迹边缘（边缘检测 → 加深）
    edges = img.filter(ImageFilter.FIND_EDGES).convert("L")
    edge_arr = np.array(edges)
    edge_mask = edge_arr > 30

    arr = np.array(img)
    # 边缘像素加深
    arr[edge_mask, :3] = (arr[edge_mask, :3] * 0.6).astype(np.uint8)
    img = Image.fromarray(arr)

    return img

def pixel_ink_pipeline(input_path: str, output_path: str, pixel_size: int = 8):
    """完整管线：像素化 → 水墨化 → 保存"""
    img = Image.open(input_path).convert("RGBA")
    img = pixelize(img, pixel_size)
    img = ink_wash_effect(img)
    img.save(output_path, "PNG")
    print(f"  ✓ {Path(input_path).name} → 像素水墨处理完成 ({output_path})")

# 批量处理
if __name__ == '__main__':
    input_dir = Path("generated_assets/raw")
    output_dir = Path("generated_assets/processed")
    output_dir.mkdir(parents=True, exist_ok=True)

    for p in input_dir.glob("*.png"):
        pixel_ink_pipeline(str(p), str(output_dir / p.name), pixel_size=8)
        # 同时生成2x版本（大像素可做备用）
        pixel_ink_pipeline(str(p), str(output_dir / f"{p.stem}_hd{p.suffix}"), pixel_size=4)

四、像素水墨风（Pixel Ink-Wash）风格指南

4.1 风格定义

像素水墨风 = 像素艺术（Pixel Art）的网格约束 + 水墨画（Ink Wash）的笔触质感。

不是简单地把水墨画分辨率降低，而是在像素的离散网格上模拟毛笔的「飞白」「晕染」「枯笔」效果。

维度像素艺术水墨画像素水墨（融合）
颜色有限色板，平涂黑白灰，灰度渐变8-16色有限色板 + 水墨灰度
线条硬边锯齿飞白、枯笔、粗细变化锯齿模拟飞白（用像素抖动表现笔触）
纹理无/纯色填色宣纸纹理、墨迹扩散像素级抖动（Dithering）模拟墨迹晕染
阴影硬边暗面泼墨渐变递进式像素渐变（1px 步进）
背景纯色/透明留白宣纸米白底色 + 大片留白

4.3 像素水墨渲染示例

// —————————————— style/InkRenderer.ts ——————————————
// Phaser 中的像素水墨渲染器

export class InkRenderer {
  /** 使用 CanvasTexture 绘制像素水墨风棋盘格子 */
  static drawInkCell(
    canvas: Phaser.Textures.CanvasTexture,
    col: number, row: number,
    cellSize: number
  ): void {
    const ctx = canvas.context;
    const x = col * cellSize, y = row * cellSize;

    // 像素风格的墨色格子边框（用1px硬边）
    ctx.strokeStyle = INK_PALETTE.INK_MID;
    ctx.lineWidth = 1;

    // 用像素模拟「毛笔顿笔」效果：角落加粗
    // 左上角
    ctx.fillStyle = INK_PALETTE.INK_DARK;
    ctx.fillRect(x, y, 3, 3);
    // 右下角
    ctx.fillRect(x + cellSize - 3, y + cellSize - 3, 3, 3);

    // 抖动（Dithering）模拟墨迹晕染
    this.drawInkSplatter(ctx, x + cellSize / 2, y + cellSize / 2, cellSize * 0.08);
  }

  /** 像素级墨迹抖动效果 */
  private static drawInkSplatter(
    ctx: CanvasRenderingContext2D,
    cx: number, cy: number, radius: number
  ): void {
    // 使用 Bayer 矩阵阈值抖动
    const bayer4x4 = [
      [0, 8, 2, 10],
      [12, 4, 14, 6],
      [3, 11, 1, 9],
      [15, 7, 13, 5],
    ];

    const r = Math.ceil(radius);
    for (let dy = -r; dy <= r; dy++) {
      for (let dx = -r; dx <= r; dx++) {
        const dist = Math.sqrt(dx * dx + dy * dy);
        if (dist > radius) continue;

        // 用 Bayer 矩阵决定是否画这个像素
        const threshold = bayer4x4[(dy + r) % 4][(dx + r) % 4] / 16;
        const density = 1 - (dist / radius);

        if (density > threshold) {
          const alpha = Math.floor(density * 200);
          ctx.fillStyle = `rgba(50, 45, 40, ${alpha})`;
          ctx.fillRect(
            Math.floor(cx) + dx,
            Math.floor(cy) + dy,
            1, 1
          );
        }
      }
    }
  }
}

4.4 像素水墨动画原则

原则说明代码实现
帧数精简 像素动画通常 4-8 帧即可，不需要60fps平滑 角色动画 4方向×3帧行走 + 1帧待机
关键帧强调 用「顿帧」和「拉伸」代替缓动曲线 攻击动画：蓄力(hold 3帧)→爆发(1帧)→恢复
墨迹粒子 粒子系统使用 INK_PALETTE 色板，低帧率渲染 Phaser particle emitter, 帧率锁定 15fps
像素级变换 移动/缩放限制为整数像素（避免sub-pixel模糊） Math.round(x) / Math.floor(y)
无半透明过度 像素风没有半透明渐变，只有硬过渡或抖动 alpha 只设 0 或 255，过渡用棋盘抖动 
// —————————————— 像素动画示例（攻击动作）——————————————
class PixelAnimation {
  update() {
    // 锁定为整数坐标，避免子像素模糊
    this.sprite.x = Math.round(this.targetX);
    this.sprite.y = Math.floor(this.targetY);

    // 攻击帧：3帧蓄力 → 1帧爆发 → 2帧恢复
    const frameMap = [0, 0, 0, 1, 2, 2]; // 6帧循环
    const frameIdx = frameMap[this.animFrame % frameMap.length];
    this.sprite.setFrame(frameIdx);
  }
}
